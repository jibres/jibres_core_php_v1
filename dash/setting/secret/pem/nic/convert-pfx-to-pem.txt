# We can extract the private key form a PFX to a PEM file with this command:
openssl pkcs12 -in ji128-june.pfx -nocerts -out irnic-ji128-june-key.pem -nodes

# Exporting the certificate only:
openssl pkcs12 -in ji128-june.pfx -clcerts -nokeys -out irnic-ji128-june-cert.pem

# Removing the password from the extracted private key:
openssl rsa -in irnic-ji128-june-key.pem -out irnic-ji128-june-key-server.key
